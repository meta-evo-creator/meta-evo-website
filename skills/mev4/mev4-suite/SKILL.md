### 使用示例
```javascript
// 应用MEV4套装进行完整任务处理
const mev4Suite = require('mev4-suite');

async function processComplexTask(taskData) {
  // 1. mev-sense：感知任务需求和环境
  const senseResult = await mev4Suite.mevSense.perceive({
    task: taskData.task,
    context: taskData.context,
    user: taskData.user
  });
  
  // 2. mev-think：分析问题并制定策略
  const thinkResult = await mev4Suite.mevThink.analyze({
    perception: senseResult,
    constraints: taskData.constraints,
    objectives: taskData.objectives
  });
  
  // 3. mev-optimize：优化执行计划
  const optimizeResult = await mev4Suite.mevOptimize.optimize({
    strategy: thinkResult.strategy,
    resources: taskData.resources,
    timeline: taskData.timeline
  });
  
  // 4. mev-evolve：从经验中学习改进
  const evolveResult = await mev4Suite.mevEvolve.learn({
    task: taskData.task,
    execution: optimizeResult.execution,
    outcomes: taskData.outcomes
  });
  
  // 5. 综合结果和持续进化
  const finalResult = mev4Suite.generateResult({
    sense: senseResult,
    think: thinkResult,
    optimize: optimizeResult,
    evolve: evolveResult
  });
  
  return finalResult;
}
```
